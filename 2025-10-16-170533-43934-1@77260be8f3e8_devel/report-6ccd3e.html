413 *y = G_MININT(-2147483647 -1);
414 return;
415 }
416
417 *x = rect.x;
418 *y = rect.y;
419 if (coord_type == ATK_XY_WINDOW)
420 {
421 cdk_window_get_origin (window, &x_toplevel, &y_toplevel);
422 *x -= x_toplevel;
423 *y -= y_toplevel;
424 }
425}
