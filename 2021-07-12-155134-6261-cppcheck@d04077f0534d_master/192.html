/*
 *--------------------------------------------------------------
 *
 * split_segment --
 *
 *      This procedure is called before adding or deleting
 *      segments.  It does three things: (a) it finds the segment
 *      containing iter;  (b) if there are several such
 *      segments (because some segments have zero length) then
 *      it picks the first segment that does not have left
 *      gravity;  (c) if the index refers to the middle of
 *      a segment then it splits the segment so that the
 *      index now refers to the beginning of a segment.
 *
 * Results:
 *      The return value is a pointer to the segment just
 *      before the segment corresponding to iter (as
 *      described above).  If the segment corresponding to
 *      iter is the first in its line then the return
 *      value is NULL.
 *
 * Side effects:
 *      The segment referred to by iter is split unless
 *      iter refers to its first character.
 *
 *--------------------------------------------------------------
 */

CtkTextLineSegment*
ctk_text_line_segment_split (const CtkTextIter *iter)
{
  CtkTextLineSegment *prev, *seg;
  CtkTextBTree *tree;
  CtkTextLine *line;
  int count;

  line = _ctk_text_iter_get_text_line (iter);
  tree = _ctk_text_iter_get_btree (iter);

  count = ctk_text_iter_get_line_index (iter);

  if (CTK_DEBUG_CHECK (TEXT))
    _ctk_text_iter_check (iter);
  
  prev = NULL;
  seg = line->segments;

  while (seg != NULL)
    {
      if (seg->byte_count > count)
        {
          if (count == 0)
            {
              return prev;
            }
          else
            {
              g_assert (count != seg->byte_count);
              g_assert (seg->byte_count > 0);

              _ctk_text_btree_segments_changed (tree);

              seg = (*seg->type->splitFunc)(seg, count);

              if (prev == NULL)
                line->segments = seg;
              else
                prev->next = seg;

              return seg;
            }
        }
      else if ((seg->byte_count == 0) && (count == 0)
               && !seg->type->leftGravity)
        {
          return prev;
        }

      count -= seg->byte_count;
      prev = seg;
      seg = seg->next;
    }
  g_error ("split_segment reached end of line!");
  return NULL;
}

#define CSEG_SIZE(chars) ((unsigned) (G_STRUCT_OFFSET (CtkTextLineSegment, body) \
        + 1 + (chars)))
#define TSEG_SIZE ((unsigned) (G_STRUCT_OFFSET (CtkTextLineSegment, body) \
        + sizeof (CtkTextToggleBody)))

/*

static void
char_segment_self_check (CtkTextLineSegment *seg)
{
  /* This function checks the segment itself, but doesn't
     assume the segment has been validly inserted into
     the btree. */

  g_assert (seg != NULL);

  if (seg->byte_count <= 0)
    {
      g_error ("segment has size <= 0");
    }

  if (strlen (seg->body.chars) != seg->byte_count)
    {
      g_error ("segment has wrong size");
    }

  if (g_utf8_strlen (seg->body.chars, seg->byte_count) != seg->char_count)
    {
      g_error ("char segment has wrong character count");
    }
}

CtkTextLineSegment*
_ctk_char_segment_new (const gchar *text, guint len)
{
  CtkTextLineSegment *seg;

  g_assert (ctk_text_byte_begins_utf8_char (text));

  seg = g_slice_alloc (CSEG_SIZE (len));
  seg->type = (CtkTextLineSegmentClass *)&ctk_text_char_type;
  seg->next = NULL;
  seg->byte_count = len;
  memcpy (seg->body.chars, text, len);
  seg->body.chars[len] = '\0';

  seg->char_count = g_utf8_strlen (seg->body.chars, seg->byte_count);

  if (CTK_DEBUG_CHECK (TEXT))
    char_segment_self_check (seg);

  return seg;
}

CtkTextLineSegment*
_ctk_char_segment_new_from_two_strings (const gchar *text1, 
					guint        len1, 
					guint        chars1,
                                        const gchar *text2, 
					guint        len2, 
					guint        chars2)
{
  CtkTextLineSegment *seg;

  g_assert (ctk_text_byte_begins_utf8_char (text1));
  g_assert (ctk_text_byte_begins_utf8_char (text2));

  seg = g_slice_alloc (CSEG_SIZE (len1+len2));
  seg->type = &ctk_text_char_type;
  seg->next = NULL;
  seg->byte_count = len1 + len2;
  memcpy (seg->body.chars, text1, len1);
  memcpy (seg->body.chars + len1, text2, len2);
  seg->body.chars[len1+len2] = '\0';

  seg->char_count = chars1 + chars2;

  if (CTK_DEBUG_CHECK (TEXT))
    char_segment_self_check (seg);

  return seg;
}

static void
_ctk_char_segment_free (CtkTextLineSegment *seg)
{
  if (seg == NULL)
    return;

  g_assert (seg->type == &ctk_text_char_type);

  g_slice_free1 (CSEG_SIZE (seg->byte_count), seg);
}

/*
 *--------------------------------------------------------------
 *
 * char_segment_split_func --
 *
 *      This procedure implements splitting for character segments.
 *
 * Results:
 *      The return value is a pointer to a chain of two segments
 *      that have the same characters as segPtr except split
 *      among the two segments.
 *
 * Side effects:
 *      Storage for segPtr is freed.
 *
 *--------------------------------------------------------------
 */

static CtkTextLineSegment *
char_segment_split_func (CtkTextLineSegment *seg, int index)
{
  CtkTextLineSegment *new1, *new2;

  g_assert (index < seg->byte_count);

  if (CTK_DEBUG_CHECK (TEXT))
    {
      char_segment_self_check (seg);
    }

  new1 = _ctk_char_segment_new (seg->body.chars, index);
  new2 = _ctk_char_segment_new (seg->body.chars + index, seg->byte_count - index);

  g_assert (ctk_text_byte_begins_utf8_char (new1->body.chars));
  g_assert (ctk_text_byte_begins_utf8_char (new2->body.chars));
  g_assert (new1->byte_count + new2->byte_count == seg->byte_count);
  g_assert (new1->char_count + new2->char_count == seg->char_count);

  new1->next = new2;
  new2->next = seg->next;

  if (CTK_DEBUG_CHECK (TEXT))
    {
      char_segment_self_check (new1);
      char_segment_self_check (new2);
    }

  _ctk_char_segment_free (seg);
  return new1;
}

/*
 *--------------------------------------------------------------
 *
 * char_segment_cleanup_func --
 *
 *      This procedure merges adjacent character segments into
 *      a single character segment, if possible.
 *
 * Arguments:
 *      segPtr: Pointer to the first of two adjacent segments to
 *              join.
 *      line:   Line containing segments (not used).
 *
 * Results:
 *      The return value is a pointer to the first segment in
 *      the (new) list of segments that used to start with segPtr.
 *
 * Side effects:
 *      Storage for the segments may be allocated and freed.
 *
 *--------------------------------------------------------------
 */

        /* ARGSUSED */
static CtkTextLineSegment *
char_segment_cleanup_func (CtkTextLineSegment *segPtr, CtkTextLine *line)
{
  CtkTextLineSegment *segPtr2, *newPtr;

  if (CTK_DEBUG_CHECK (TEXT))
    char_segment_self_check (segPtr);

  segPtr2 = segPtr->next;
  if ((segPtr2 == NULL) || (segPtr2->type != &ctk_text_char_type))
    {
      return segPtr;
    }

  newPtr =
    _ctk_char_segment_new_from_two_strings (segPtr->body.chars, 
					    segPtr->byte_count,
					    segPtr->char_count,
                                            segPtr2->body.chars, 
					    segPtr2->byte_count,
					    segPtr2->char_count);

  newPtr->next = segPtr2->next;

  if (CTK_DEBUG_CHECK (TEXT))
    char_segment_self_check (newPtr);

  _ctk_char_segment_free (segPtr);
  _ctk_char_segment_free (segPtr2);
  return newPtr;
}

CtkTextLineSegment*
_ctk_toggle_segment_new (CtkTextTagInfo *info, gboolean on)
{
  CtkTextLineSegment *seg;

  seg = g_slice_alloc (TSEG_SIZE);

  seg->type = on ? &ctk_text_toggle_on_type : &ctk_text_toggle_off_type;

  seg->next = NULL;

  seg->byte_count = 0;
  seg->char_count = 0;

  seg->body.toggle.info = info;
  seg->body.toggle.inNodeCounts = 0;

  return seg;
}

void
_ctk_toggle_segment_free (CtkTextLineSegment *seg)
{
  if (seg == NULL)
    return;

  g_assert (seg->type == &ctk_text_toggle_on_type ||
            seg->type == &ctk_text_toggle_off_type);

  g_slice_free1 (TSEG_SIZE, seg);
}

toggle_segment_delete_func (CtkTextLineSegment *segPtr, CtkTextLine *line, int treeGone)

      _ctk_toggle_segment_free (segPtr);
      return 0;

    }
  return 1;
}
